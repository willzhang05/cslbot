[docs]def gen_nato(msg):
    nato = ""
    for x in msg:
        x = x.upper()
        if x in nato_codes:
            nato += "%s " % nato_codes[x]
        else:
            nato += "? "
    return nato

[docs]def cmd(send, msg, args):
    """Converts text into NATO form.
    Syntax: {command} <text>
    """
    if not msg:
        send("NATO what?")
        return
    nato = gen_nato(msg)
    if len(nato) > 100:
        send("Your NATO is too long. Have you considered letters?")
    else:
        send(nato)
